    # LuckyBing 中秋博饼

    中秋博饼 Demo

    # 中秋博饼 Demo：

    博中信息
    恭喜你博中
    剩余次数：100
    <lee-luck-bing ref="LuckBing" :voice="voice" v-model="luckyNum" @start="startCallBack" @end="endCallBack"></lee-luck-bing>  

    
<script>
    export default {
        data() {
              return {
               luckyNum:13,
               logo:'/bobing/sz/wyc.png',
               voice:{
               yao:'/bobing/yao.mp3',
               zero:'/bobing/zero.mp3',
               one:'/bobing/one.mp3',
               two:'/bobing/two.mp3',
               three:'/bobing/three.mp3',
               four:'/bobing/four.mp3',
               five:'/bobing/five.mp3',
               six:'/bobing/six.mp3'
               }
              }
          },
          methods:{
            startCallBack(){
              if (!this.luckyNum) {
              alert('还剩0次机会')
              return 
              }
              let arr=[]
              for(let i=0;i<6;i++){
               arr[i]=Math.ceil(Math.random() * 6)               
              }
              this.$refs.LuckBing.play(arr)
            },
            endCallBack(rs){
              alert('恭喜你博中:'+rs)
              this.luckyNum--
            }
          }
    }
</script>
