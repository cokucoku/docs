    # Badge 徽章

    # 示例：

    站内信息8 购物车99
    <lee-badge title="站内信息" info="8"/>
<lee-badge title="购物车" info="99"  color="#fc0"/>

    # 属性

    参数 说明 类型 可选值 默认值
    title 徽章项目 String -- --
    info 徽章文字 String -- --
    href 链接地址 String -- --
    color 徽章颜色 String -- '#f00'
