    # Cells属性

    参数 说明 类型 可选值 默认值
    title Cells的标题 String - -
    footer Cells底部的文字 String - -

    # Cell属性

    参数 说明 类型 可选值 默认值
    title 最左侧的标题 String - -
    footer Cell右侧区域的内容 String - -
    icon 最左侧的icon String - -
    url 跳转地址 String - -
    del 是否可以删除 Boolean - false
