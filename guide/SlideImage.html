    # SlideImage 滑动轮播图

    # 基础示例：

    <lee-slide-image v-model="show"  :config="config" ref="slideimg">
</lee-slide-image>

    <script>
    export default {
        data() {
         return {
            show: false,
            config:[]
           
        }        
       },
       mounted() {
         let img = document.querySelectorAll('.slide')
         img.forEach((el) => {
            this.config.push(el.getAttribute('src'))
        })
       },
       methods: {
        go(xh){
        this.show=true       
        this.$refs.slideimg.slideto(xh) 
        }
       }
    }
</script>
