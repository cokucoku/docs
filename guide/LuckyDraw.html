        data() {

                  {fonts:{text:'10元红包',top:'80%'},imgs: {src:'/img/zp-4.png',width:'25%',top:'70%'},background:'#f9e3bb'},
                  {fonts:{text:'50元红包',top:'80%'},imgs: {src:'/img/zp-5.png',width:'25%',top:'70%'},background:'#f8d384'},
                  {fonts:{text:'0.3元红包',top:'80%'},imgs: {src:'/img/zp-6.png',width:'25%',top:'70%'},background:'#f9e3bb'},
                  {fonts:{text:'5元红包',top:'80%'},imgs: {src:'/img/zp-7.png',width:'25%',top:'70%'},background:'#f8d384'}
                ],
               config:{                
                 width:350,
                 height:350,
                 padding:20,
                 fontSize:'16px',
                 fontColor:'#d64737',
                 backgroundColor: '#d64737',             
               },
               btnConfig:{                   
                width:'80',
                background: '#d64737',
                fonts: [
                  { text: '开始', fontColor: '#fff', top: '50%', fontSize: '25px' },
                  { text: '消费10积分', fontColor: '#fff', top: '25%', fontSize: '11px' },
                ]                  
               },
              }
          },
          methods:{
            startCallBack(){
              if (!this.luckyNum) return alert('还剩0次抽奖机会')
              this.$refs.LuckDraw.play(Math.ceil(Math.random() * this.prizes.length))
            },
            endCallBack(inx){
              alert(`恭喜你获得大奖: ${this.prizes[inx].fonts.text}`)
              this.luckyNum--
            }
          }
    }
</script>

    # LuckyDraw 属性

    参数 说明 类型 可选值 默认值
    ref 转盘名 String -- --
    config 配置转盘 Object width,height,padding,fontSize,fontColor,backgroundColor,backgroundImage --
    btnConfig 配置按钮 Object width,background,fonts,imgs --
    prizes 配置礼品 Array fonts,imgs,background --

    # LuckyDraw 事件

    事件名称 说明 回调参数
    start 开始抽奖事件 --
    end 抽奖结束事件 中奖索引

    # LuckyDraw 方法

    方法名称 说明 参数
    play 抽奖方法 抽中那个索引

    # config 设置

    参数 说明 类型 可选值 默认值
    width 转盘宽度 Number -- --
    height 转盘高度 Number -- --
    padding 转盘内边距 Number -- --
    fontSize 转盘字号 String '16px' --
    fontColor 字体颜色 String '#d64737' --
    backgroundColor 转盘背景颜色 String '#d64737' --
    backgroundImage 转盘背景图 String -- --
    rotate 背景转动 Boolean true,false false
    rollNum 旋转次数 Number -- 6

    # btnConfig 设置

    参数 说明 类型 可选值 默认值
    width 按钮大小 Number -- --
    background 按钮背景色 String '#FFDCB8)' --
    fonts 按钮文字 Array {
    text: '开始',//按钮文字
    fontColor: '#1c258e',//按钮文字颜色
    top: '18%',//按钮文字顶部位置
    fontSize:'30px'//按钮文字大小
    }    		 --
    imgs 按钮图片 Array {
    src: '/active-eat.png',//按钮图片
    width: '100%',//按钮图片大小
    top: '18%',//按钮图片顶部位置
    }    		 --

    # prizes 设置

    参数 说明 类型 可选值 默认值
    background 格子背景颜色 String '#1c258e' --
    fonts 奖品文字设置 Object {
    text: '草莓蛋糕',//奖品文字
    top: '18%',//奖品文字顶部位置
    }    		 --
    imgs 奖品图片设置 Object {
    src:'/img/default-eat.png',//奖品图片
    width:'100%',//奖品图片宽度
    top:'0%'//奖品图片顶部位置
    }    		 --
