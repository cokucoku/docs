    # Stripe：

    <lee-scratch :prizes="prizes" stripe stripe></lee-scratch>
    显示代码

    # Black area：

    <lee-scratch :prizes="prizes" background="#000" :width="300" :height="150"></lee-scratch>
    显示代码

    # Brush Size：

    <lee-scratch :prizes="prizes" :size="10" textColor="#3eaf7c" background="#3eaf7c" :width="300" :height="150"></lee-scratch>
    显示代码

    # Attribute

    Parameter Description Type Optional Value Default Value
    prizes prizes Array -- --
    background background String -- #eee
    stripe stripe Boolean true,false false
    textColor textColor String -- #f20
    textSize textSize Number -- 30
    width width Number -- 260
    height height Number -- 112
    size Brush Size Number -- 20
